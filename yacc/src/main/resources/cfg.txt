LANGUAGE -> {FUNC} {LANGUAGE} | ε
FUNC -> FUNCTION ID LT {ARGS} GT LCB {STATEMENT} RCB
ARGS -> ID {ARG} | ε
ARG -> COMMA ID {ARG} | ε
STATEMENT -> {EXPRESSION} SEMICOLON {STATEMENT} | ε
EXPRESSION -> {ASSIGNMENT} | {VALUE} | {RETURN} | {BLOCK} | ε
BLOCK -> {FOR-LOOP} | {WHILE-LOOP} | {IF-ELSE}
ASSIGNMENT -> VAR ID DEF {VALUE} | VAL ID DEF {VALUE} | ID DEF {VALUE}
VALUE -> INT | FLOAT | {CALL}
CALL -> ID {CALL-ARG}
CALL-ARG -> LSB {VALUE} RSB | LT {ARGS} GT | INC | DEC | {BINARY-OP} {VALUE} | ε
BINARY-OP -> ADD | MINUS | MUL | DIV | MOD | DEF | EQU | LE | GT | GE
FOR-LOOP -> FOR {FOR-COND} LCB {STATEMENT} RCB
FOR-COND -> {ASSIGNMENT} SEMICOLON {VALUE} SEMICOLON {VALUE}
WHILE-LOOP -> WHILE {VALUE} LCB {STATEMENT} RCB
IF-ELSE -> IF {VALUE} LCB {STATEMENT} RCB ELSE LCB {STATEMENT} RCB
RETURN -> RET {VALUE}
