LANGUAGE -> FUNC {LANGUAGE} | ε
FUNC -> FUNCTION ID LT {ARGS} GT LCB {STATEMENT} RCB
ARGS -> ID {ARG} | ε
ARG -> COMMA ID {ARG} | ε
STATEMENT -> {EXPRESSION} COMMA {STATEMENT} | ε
EXPRESSION -> {ASSIGNMENT} | {CALCULATION} | {FOR} | {WHILE} | {IF-ELSE} | {RETURN} | ε
ASSIGNMENT -> VAR ID DEF {VALUE} | VAL ID DEF {VALUE}
VALUE -> ID | INT | FLOAT | {LIST} | {CALL} | {CALCULATION}
LIST -> LSB {ELEMENTS} RSB
ELEMENTS -> {VALUE} {ELEMENT} | ε
ELEMENT -> COMMA {VALUE} | ε
ELEMENT-VALUE -> ID LSB {VALUE} RSB
CALL -> ID LT {ARGS} GT
CALCULATION -> {UNARY-CAL} | {BINARY-CAL}
UNARY-CAL -> INC ID | ID INC | DEC ID | ID DEC
BINARY-CAL -> {VALUE} {BINARY-OP} {RIGHT}
BINARY-OP -> ADD | MINUS | MUL | DIV | MOD | EQU | LE | LT | GT | GE |
FOR-LOOP -> FOR {FOR-COND} LCB {STATEMENT} RCB
FOR-COND ->  ID IN ID | {CALCULATION} SEMICOLON {CALCULATION} SEMICOLON {CALCULATION}
WHILE-LOOP -> WHILE {CALCULATION} LCB {STATEMENT} RCB
IF-ELSE -> IF {STATEMENT} LCB {STATEMENT} RCB ELSE LCB {STATEMENT} RCB
RETURN -> RET {VALUE}
